import os 
import codecs 

def writefile(fn, v_ls): 
    f = codecs.open(fn, 'wb', 'utf-8') 
    for i in v_ls: 
        f.write(i + os.linesep) 
    f.close() 

def readfile(fn): 
    f = codecs.open(fn,'r','utf-8') 
    ls = [ line.strip() for line in f] 
    f.close() 
    for i in ls: 
        print i
